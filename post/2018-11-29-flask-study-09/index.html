Flask Study 09 上传图片基础

2018-11-29 Liseen

人生苦短,我学Python…上传图片在web应用中十分的常见,上传个身份证审核什么的.

文件夹结构如下:

├── app.py    # 入口
├── forms.py  # Form表单
├── static    # 静态资源
│   ├── css
│   ├── favicon.ico
│   ├── js
│   └── style.css
├── templates   # 模板文件夹
│   ├── base.html
│   ├── bootstrap.html
│   ├── index.html
│   ├── macros.html
│   ├── showimages.html
│   └── upload.html
└── uploads   # 服务器文件存储目录
    ├── 03b82ee6f55d415b857cdf9ddfa2502f.png

app.python

import os
# 唯一码
import uuid
from flask import Flask,render_template,redirect,url_for,flash,session,send_from_directory
from forms import LoginForm,UploadForm
app = Flask(__name__)

app.secret_key = 'dev secret'
# 定义上传目录
app.config['UPLOAD_PATH'] = os.path.join(app.root_path, 'uploads')

def random_filename(filename):
    ext = os.path.splitext(filename)[1]
    new_filename = uuid.uuid4().hex + ext
    return new_filename

# 获取上传文件
@app.route('/uploads/<path:filename>')
def get_file(filename):
    return send_from_directory(app.config['UPLOAD_PATH'], filename)
# 显示上传图片视图
@app.route('/show_images')
def show_images():
    return render_template('showimages.html')
# 上传Form视图
@app.route('/upload',methods=['GET','POST'])
def upload():
    form = UploadForm()
    if form.validate_on_submit():
        file = form.photo.data
        filename = random_filename(file.filename)
        file.save(os.path.join(app.config['UPLOAD_PATH'],filename))
        flash('Upload success.')
        session['filenames'] = [filename]
        return redirect(url_for('show_images'))
    return render_template('upload.html',form=form)

{% macro form_field(field) %}
    {{ field.label }}<br>
    {{ field(**kwargs) }}<br>
    {% if field.errors -%}
        {% for error in field.errors %}

{% endmacro %}
